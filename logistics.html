import os import json import math from datetime import datetime # ========================= # CONFIG # ========================= BASE_URL = "https://brightlane.github.io/SameDayFlowers" PROJECT_DIR = "blog" SITEMAP_DIR = "sitemaps" AFFILIATE_ID = "2013017799" AFF_URL = f"https://www.floristone.com/main.cfm?AffiliateID={AFFILIATE_ID}&source_id=brightlane" CHUNK_SIZE = 45000 TODAY = datetime.now().strftime("%Y-%m-%d") # ========================= # SAFE HOOK GENERATOR FIXED # ========================= ACTIONS = ["Send", "Buy", "Order", "Deliver", "Get", "Ship"] SUBJECTS = ["Luxury Bouquets", "Premium Roses", "Mother's Day Flowers", "Spring Tulips", "Designer Arrangements"] URGENCY = ["Express", "Same-Day", "Today", "Now", "Guaranteed"] HOOKS = [ f"{a} {s} in {{city}} - {{urgency}}" for a in ACTIONS for s in SUBJECTS for _ in URGENCY ] # ========================= # HELPERS # ========================= def ensure_dirs(): os.makedirs(PROJECT_DIR, exist_ok=True) os.makedirs(SITEMAP_DIR, exist_ok=True) def slug(city, state): return f"{city.lower().replace(' ', '-')}-{state.lower()}" # ========================= # PAGE GENERATOR # ========================= def generate_page(city, state): hook = random.choice(HOOKS).format( city=city, urgency=random.choice(URGENCY) ) filename = f"flowers-{slug(city, state)}.html" path = os.path.join(PROJECT_DIR, filename) html = f""" {city} Flower Delivery

{hook}

Same-day delivery in {city}, {state}

Order Flowers """ with open(path, "w", encoding="utf-8") as f: f.write(html) return filename # ========================= # SITEMAP CHUNKS # ========================= def generate_sitemaps(cities): chunks = math.ceil(len(cities) / CHUNK_SIZE) files = [] for i in range(chunks): part = cities[i * CHUNK_SIZE:(i + 1) * CHUNK_SIZE] name = f"part-{i+1}.xml" path = os.path.join(SITEMAP_DIR, name) with open(path, "w", encoding="utf-8") as f: f.write('\n') f.write('\n') for c in part: url = f"{BASE_URL}/{PROJECT_DIR}/flowers-{slug(c['city'], c['state'])}.html" f.write(f"{url}{TODAY}\n") f.write("") files.append(name) return files # ========================= # SITEMAP INDEX # ========================= def generate_sitemap_index(files): path = "sitemap.xml" with open(path, "w", encoding="utf-8") as f: f.write('\n') f.write('\n') for file in files: f.write("\n") f.write(f"{BASE_URL}/{SITEMAP_DIR}/{file}\n") f.write(f"{TODAY}\n") f.write("\n") f.write("") # ========================= # LLMs.TXT GENERATOR # ========================= def generate_llms(cities): path = "llms.txt" content = f"""# SameDayFlowers AI Index Affiliate ID: {AFFILIATE_ID} ## Top Cities """ for c in cities[:500]: url = f"{BASE_URL}/{PROJECT_DIR}/flowers-{slug(c['city'], c['state'])}.html" content += f"- {c['city']}, {c['state']} → {url}\n" content += f"\n## Affiliate Link\n{AFF_URL}\n" with open(path, "w", encoding="utf-8") as f: f.write(content) # ========================= # SCHEMA.ORG GENERATOR # ========================= def generate_schema(): schema = { "@context": "https://schema.org", "@type": "Florist", "name": "SameDayFlowers", "url": BASE_URL, "telephone": "+1-800-FLORIST", "priceRange": "$$", "areaServed": ["US", "CA"], "potentialAction": { "@type": "OrderAction", "target": AFF_URL } } with open("schema.json", "w", encoding="utf-8") as f: json.dump(schema, f, indent=2) # ========================= # MAIN PIPELINE # ========================= if __name__ == "__main__": ensure_dirs() with open("cities.json", "r", encoding="utf-8") as f: cities = json.load(f) print("🚀 Generating Pages...") for c in cities[:2000]: generate_page(c["city"], c["state"]) print("🗺 Generating Sitemaps...") parts = generate_sitemaps(cities) print("📌 Generating Sitemap Index...") generate_sitemap_index(parts) print("🤖 Generating llms.txt...") generate_llms(cities) print("🧬 Generating Schema.org...") generate_schema() print("✅ BUILD COMPLETE — Affiliate ID ACTIVE:", AFFILIATE_ID)
